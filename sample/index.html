zip2address sample

var input = $('input'),
    debug = $('#debug');

$('button').click(function() {
    zip2address(input.val(), function(address) {
        if (address) {
            debug.html([
                '住所: '     + address.all,
                '都道府県: ' + address.pref,
                '市町村: '   + address.city
            ].join('<br>'));
        }
        else {
            debug.html('取得に失敗しました');
        }
    });
});
